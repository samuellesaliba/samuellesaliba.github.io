// the setup function runs once when you press reset or power the board void setup() { // initialize digital pin LED_BUILTIN as an output. pinMode(LED_BUILTIN, OUTPUT); //red LED pinMode(7, OUTPUT); //yellow LED pinMode(10, OUTPUT); //green LED } // the loop function runs over and over again forever void loop() { digitalWrite(LED_BUILTIN, HIGH); // turn the red LED on delay(500); // wait for a 50ms digitalWrite(7, HIGH); // turn the yellow LED on delay(500); // wait for 50ms digitalWrite(10, HIGH); // turn the green LED on delay(1000); // wait for a second digitalWrite(10, LOW); // turn the red LED on delay(500); // wait for a 50ms digitalWrite(7, LOW); // turn the yellow LED on delay(500); // wait for 50ms digitalWrite(LED_BUILTIN, LOW); // turn the green LED on delay(500); // wait for a second }
